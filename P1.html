        let literalObjectName = {
            "key1":"value1",
            "key2":"value2",
            "key3":"value3",
            "key4":"value4",
        };

        console.log(literalObjectName);
        console.log(typeof(literalObjectName));
        console.log(literalObjectName.key1);//dispalys key value
